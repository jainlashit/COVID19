	\section{SIDS}
	\begin{equation*}
		\dv{s}{t} = -\beta si + \alpha i
	\end{equation*}
	\begin{equation*}
		\dv{i}{t} = \beta si - \alpha i - \gamma i
	\end{equation*}
	\begin{equation*}
		\dv{d}{t} = \gamma i
	\end{equation*}
	where $\gamma$ represents mortality rate, $\beta$ represents transmission rate and $\alpha$ represents rate of true recovery (not dying).
